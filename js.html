Q.Check whether a Number is a prime or not.

const Prime_Check = (N) => {
	let i;
	let isPrime = true;
	  if(N===0 || N===1) {
	  isPrime = false;
	}
	else {
	  for(i=2 ; i<=N/2 ; i++){
		if (N%i === 0){
		  isPrime = false;
		  break;
		}
		else {
		  isPrime = true;
		}
	  }
	}
	if(isPrime)
	return("YES");
	else
	return("NO");
  };
